Sum_Dziel_Wlasciwych(n)
	suma = 1
	for i=2 to n-1 do
        if n mod i == 0
            then suma=suma+i
    return suma

Max_Min_dziel(tab[1..n])
    max = tab[1]
    min = tab[1]
    for i=2 to n do
        if max < tab[i]
            then max = tab[i]
        if min > tab[i] 
            then min = tab[i]
    return max,min

BinarySearch(tab[1..n], x)
    min = 1
    max = n
    while min =< max do
        avg = (min+max) div 2
        if tab[avg] > x then
            max = avg
        else if tab[avg] < x then 
            min = avg
        else then
            return true
    return false

Fibonacci(n)
    fib = 0
    pierwsza = 0
    druga = 1
    if n == 1 then 
        return 0
    if n == 2 then
        return 1
    for i = 3 to n do
        fib = pierwsza + druga
        pierwsza = druga
        druga = fib
    return fib
